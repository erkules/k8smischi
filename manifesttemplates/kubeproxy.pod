apiVersion: v1
kind: Pod
metadata:
  name:      kube-proxy
  namespace: kube-system
spec:
  hostNetwork: true
  containers:
  - name: kube-proxy
    image: "{{ hyperkube }}"
    command:
    - /hyperkube
    - kube-proxy
    - --kubeconfig=/etc/kubernetes/ssl/kubeconfig.yml
    - --master={% for host in groups['worker'][0] -%}https://{{hostvars[host]['ansible_ssh_host'] }}:6443 {% endfor -%}
    securityContext:
      privileged: true
    volumeMounts:
    - mountPath: /etc/kubernetes/ssl
      name: ssl
  volumes:
  - name: ssl
    hostPath:
      path: /etc/kubernetes/ssl
